PROGRAM S2017;
//This program computes and prints the value
	//Of an expression//

  //declare variables//
INTEGER: P1,	Q2S,	R, S12 ;
BEGIN
	P1		= -3 ;
	Q2S =	4 ;
	R		= 5  ;

	//compute the value of the expression //
  S12 = P1 * ( Q2S + 2 * R)		;

END.
